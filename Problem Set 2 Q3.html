from pandas.plotting import scatter_matrix
file_path = 'COL-1.csv'  
data = pd.read_csv(file_path)
numeric_cols = data.select_dtypes(include=['float64', 'int64'])
scatter_matrix(numeric_cols, figsize=(12, 8), diagonal='kde', alpha=0.8)
plt.suptitle('Scatterplot Matrix of Numeric Columns')
plt.show()

#e) Choose a single subplot that seems most interesting to you and make a separate scatterplot of just those two columns with the points colored by the salary value.

plt.figure(figsize=(8,6))

            c=data['Avg Disposable Income'], cmap='viridis', s=50, alpha=0.7)
